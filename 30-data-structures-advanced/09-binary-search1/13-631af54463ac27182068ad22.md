### Q7. What is the output of the below code?
#include <iostream>
int bound(int arr[], int N, int X)
{
	int mid;
	int low = 0;
	int high = N;
	while (low < high) {
		mid = low + (high - low) / 2;
		if (X >= arr[mid]) {
			low = mid + 1;
		}
		else {
			high = mid;
		}
	}
	if(low < N && arr[low] <= X) {
		low++;
	}
	return low;
}
int main()
{
	int arr[] = { 4, 6, 10, 12, 18, 20 };
	int N = sizeof(arr) / sizeof(arr[0]);
	std::cout << bound(arr, N, 15);
	return 0;
}
