<h3>Q7. What is the output of the below code?
#include &lt;iostream&gt;
int bound(int arr[], int N, int X)
{
	int mid;
	int low = 0;
	int high = N;
	while (low &lt; high) {
		mid = low + (high - low) / 2;
		if (X &gt;= arr[mid]) {
			low = mid + 1;
		}
		else {
			high = mid;
		}
	}
	if(low &lt; N &amp;&amp; arr[low] &lt;= X) {
		low++;
	}
	return low;
}
int main()
{
	int arr[] = { 4, 6, 10, 12, 18, 20 };
	int N = sizeof(arr) / sizeof(arr[0]);
	std::cout &lt;&lt; bound(arr, N, 15);
	return 0;
}</h3>
