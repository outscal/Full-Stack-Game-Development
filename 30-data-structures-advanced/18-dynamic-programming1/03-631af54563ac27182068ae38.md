<h3>Q6. Predict the output: 
#include &lt;bits/stdc++.h&gt;
using namespace std;
int solve(int n) {
  int i, j, arr[n], k;
  arr[0] = 1;
  for (i = 1; i &lt; n; i++) {
    arr[i] = 0;
    for (j = 0, k = i - 1; j &lt; i; j++, k--)
      arr[i] += arr[j] * arr[k];
  }
  return arr[n - 1];
}
int main() {
