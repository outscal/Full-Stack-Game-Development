### Q3. What will be the output of the given code?

class Pokemon {
public:
    void attack() {
        std::cout << "Pokemon uses a basic attack!";
    }
};

class FireType : public virtual Pokemon {
public:
    void attack() {
        std::cout << "FireType Pokemon uses Fire Attack!";
    }
};

class FlyingType : public virtual Pokemon {
public:
    void attack() {
        std::cout << "FlyingType Pokemon uses Flying Attack!";
    }
};

class Charizard : public FireType, public FlyingType {};

int main() {
    Charizard charizard;
    charizard.attack();
    return 0;
}

-  "FireType Pokemon uses Fire Attack!"
-  "FlyingType Pokemon uses Flying Attack!"
- Compilation error due to ambiguous attack() method **(correct)**
- "Pokemon uses a basic attack!"
