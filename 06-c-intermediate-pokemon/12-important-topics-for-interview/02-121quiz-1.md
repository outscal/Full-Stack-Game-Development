### Q1. Consider the following code:

public:
    int publicVar;
protected:

};

public:
    void display() {
        publicVar = 10;
        protectedVar = 20;
        // privateVar = 30; // Error
    }
};

int main() {
    Derived d;
    d.publicVar = 10;  // Error
    return 0;
}

- Because publicVar becomes private in the Derived class
- Because publicVar is private in the Base class

class Derived : public Base {
public:
    void show() {
        std::cout << "Derived class show" << std::endl;
    }
};

int main() {
    Derived d;
    d.display();  // Line 1
    d.show();     // Line 2
    return 0;
}

-  Both d.display() and d.show() will still work correctly
- Only d.show() will work, and d.display() will cause a compilation error **(correct)**
