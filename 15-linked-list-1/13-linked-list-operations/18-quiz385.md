### Q7. What will be the output of this program when executed?

using namespace std;

  int data;
  Node* next;
};

void traverse(Node* head) {
  while (head != NULL) {
    cout << head->data << " ";
    head = head->next; 
  }
}

int main() {
  Node* head = new Node();
  Node* second = new Node();
  Node* third = new Node();

  head->data = 1;
  head->next = second;

  second->data = 2;
  second->next = third;

  third->data = 3;
  third->next = NULL;

  while (head != NULL) {
    cout << head->data << " ";
    head = head->next; 
  }
  cout << endl;

  traverse(head);

  return 0;
}
