struct Node {
    int data;
    Node* next;
};

Node* reverseList(Node* head) {

    Node* current = head;
    Node* next = nullptr;

    while (current != nullptr) {
        next = current->next;
        current->next = prev;
        prev = current;
        current = next;

    head = prev;
    return head;
}
