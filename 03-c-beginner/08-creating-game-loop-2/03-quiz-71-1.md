### Q1. How many iterations (frames) will the game loop execute before the program exits?

using System;
using System.Threading;

class Program
{
    static void Main()
    {
        int totalTime = 5000; // Total game time in milliseconds
        int frameRate = 60;    // Frames per second
        int frameDuration = 1000 / frameRate; // Duration of each frame in milliseconds
        int currentFrame = 0;
        int elapsedTime = 0;
        bool isRunning = true;

        while (isRunning)
        {
            currentFrame++;
            elapsedTime += frameDuration;

            Console.WriteLine($"Frame: {currentFrame}, Elapsed Time: {elapsedTime}ms");

            if (elapsedTime >= totalTime)
                isRunning = false;

            Thread.Sleep(frameDuration);
        }

        Console.WriteLine("Game Loop Ended.");
    }
}

-  60 **(correct)**
-  120
- Infinite Loop

### Q5. What will be the final output if the user presses the 'Q' key during the second iteration of the game loop?

using System;
using System.Threading;

class Program
{
    static bool isRunning = true;
    static int score = 0;

    static void Main()
    {
        Console.WriteLine("Game Started. Press 'P' to pause or 'Q' to quit.");

        while (isRunning)
        {
            HandleInput();
            if (!isRunning)
                break;

            Update();
            Render();

            Thread.Sleep(500); // Pause for half a second
        }

        Console.WriteLine("Game Ended with Score: " + score);
    }

    static void HandleInput()
    {
        if (Console.KeyAvailable)
        {
            var key = Console.ReadKey(true).Key;
            if (key == ConsoleKey.Q)
                isRunning = false;
            else if (key == ConsoleKey.P)
                PauseGame();
        }
    }

    static void Update()
    {
        score += 10;
    }

    static void Render()
    {
        Console.Clear();
        Console.WriteLine("Current Score: " + score);
    }

    static void PauseGame()
    {
        Console.WriteLine("Game Paused. Press 'R' to resume.");
        while (true)
        {
            var key = Console.ReadKey(true).Key;
            if (key == ConsoleKey.R)
                break;
        }
    }
}
- Game Ended with Score: 10
- Game Ended with Score: 20 **(correct)**
- Game Ended with Score: 0
- An exception is thrown due to improper loop termination.
